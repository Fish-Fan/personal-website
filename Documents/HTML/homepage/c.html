                        using System;
                        using System.Collections.Generic;
                        using System.Linq;
                        using System.Text;
                        using System.Threading.Tasks;
                        //编写一个猜数字游戏
                        namespace guessNum
                        {
                            class Program
                                {
                                static void Main(string[] args)
                                    {
                                    Random randomNum = new Random();//声明一个新对象
                                    int rightNum = randomNum.Next(1,100);//产生一个1到100的随机数

                                    int count = 1;//计数器
                                    do
                                        {
                                        Console.WriteLine("请输入您猜的数字（1—100）");
                                        int guessNum = int.Parse(Console.ReadLine());//获取用户输入的值


                                        if (guessNum == rightNum)
                                            {
                                            Console.WriteLine("太棒了，您猜对了！正确的数就是{0}", rightNum);
                                            break;
                                            }
                                        else if (guessNum < rightNum)
                                            {
                                            Console.WriteLine("很抱歉，您猜的数小了，请继续努力哦！");
                                            }
                                        else
                                            {
                                            Console.WriteLine("很抱歉，您猜的数大了，请继续努力哦！");
                                            }
                                            count++;
                                        } while (count <= 3);
                                    Console.WriteLine("很抱歉，您已经三次未猜中正确数字了.");
                                    Console.WriteLine("正确的数字是{0}", rightNum);



                                    }
                                }
                        }
